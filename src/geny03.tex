\textbf{2. Nội dung chi tiết}

\subsecheading{2.1. Cập nhật cấu trúc mã nguồn:}
\introtext{Phúc đã thực hiện tái cấu trúc lại dự án trên nền tảng Overleaf/LaTeX để tối ưu hóa việc quản lý và đồng bộ dữ liệu:}
\begin{itemize}[leftmargin=3.3em]
    \item[o] \textbf{Thư mục assets:} Gom toàn bộ hình ảnh, chữ ký của các thành viên vào một thư mục chung để mã nguồn gọn gàng hơn.
    \item[o] \textbf{Tệp setup.tex:} Tập trung khai báo toàn bộ thư viện và định dạng văn bản (format chữ, lề, font...) vào một tệp duy nhất. Việc này đảm bảo tính nhất quán cho tất cả các phần nội dung do các thành viên khác nhau biên soạn.
    \item[o] \textbf{Tệp main.tex:} Dùng làm tệp tổng để nhúng (input) các nội dung từ các file thành phần.
\end{itemize}

\subsecheading{2.2. Hoàn thiện nội dung báo cáo:}
\begin{itemize}[leftmargin=3.3em]
    \item[o] \textbf{Trang bìa:} Phúc đã cập nhật lại trang bìa dựa trên thiết kế của Vĩ, bổ sung thêm cụm từ "Lập trình viên toàn phần" (Fullstack Developer) để làm nổi bật vị trí việc làm mà nhóm đã chọn.
    \item[o] \textbf{Tiến độ:} Các tệp nội dung của các thành viên đã được tích hợp vào bản thảo cuối cùng (Final).
\end{itemize}

\subsecheading{2.3. Quản lý dự án trên GitHub:}
